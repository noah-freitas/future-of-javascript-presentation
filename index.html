let

if (true) {

block scoped

var funcs = [];

for (let i = 0; i < 10; i++)
    funcs.push(function () { return i; });

funcs[0]() === 0

Destructuring

Destructuring

  • Object and Array literal-like syntax for assignment.
  • Declarative assignment
  • Two pieces:
    • The value being destructured/matched.
    • The pattern used for matching/destructuring.

Destruct. Objects

let { a : a } = { a : 2 };
a === 2
// Desugared:
var b = { a : 2 }, a = b.a;
a === 2
let { a } = { a : 2 };
a === 2

Destruct. Arrays

let [x] = [1];
x === 1
// Desugared:
var y = [1], x = y[0];
x === 1
let [a, , b, c] = [1, 2, 3, 4];
a === 1
b === 3
c === 4

Rest Operator

Rest Operator

  • Gathers 0+ elements from a matched list into a new array.
  • Allows for functional head/tail array destructuring.
let [head, ...tail] = [1, 2, 3, 4];
head === 1
tail === [2, 3, 4]
let [head, ...tail] = [];
head === undefined
tail === []
let [a, b, ...cs] = [1, 2, 3];
a  === 1
b  === 2
cs === [3]

Default Values

Default Values

  • Fill in missing values during destructuring.
  • Provides defaults for whole patterns or pieces of patterns.
let [{ a }] = []; // TypeError
let [{ a } = {}] = [];
a === undefined
let [{ a = 3 } = {}] = [];
a === 3
let [{ a = 3 } = {}] = undefined; // TypeError
let ([{ a = 3 } = {}] = []) = undefined;
a === 3

Dest. Parameters

  • Destructured value is the actual parameter
  • Destructuring pattern is the formal parameter
  • Each parameter can have its own pattern
  • The rest operator can destructure the parameter list itself

Dest. Parameters

function add(...nums) {
  return nums.reduce(function (x, y) { return x + y; }, 0);
}

add(1, 2, 3, 4) === 10
add()           === 0
function add(x = 0, y = 0) {
  return x + y;
}

add()  === 0;
add(1) === 1;
document.querySelector('button').addEventListener('click', alertText);

function alertText({ target : { textContent : text } }) {
  alert(text);
}

Dest. Parameters

function weirdAdd({ a : [x = 1, ...ys] }, { b } = { b : 16 }) {
  return x * 2 + ys.reduce((x, y) => x + y, 0) + Math.sqrt(b);
}

weirdAdd({ a : [1, 2, 3] }, { b : 4 }) === 9
weirdAdd({ a : [] })                   === 6
weirdAdd()                             === // TypeError
