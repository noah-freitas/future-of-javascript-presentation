block scoped

if (true) {
    var x = 3;
    let y = 3;
}

x === 3
y // ReferenceError: y is not defined

if (let x = getValue()) {
    // x === the result of `getValue()`
}

x // ReferenceError: x is not defined

lexically scoped

  • "temporal deadzone"
  • TypeErrors for redeclaration in blocks
if (true) {
    x; // ReferenceError: can't access
       // let declaration `x' before initialization

    let x = 3;

    x = 5;

    let x = 4; // TypeError: redeclaration of variable x
    var x = 5; // TypeError: redeclaration of let x
}
