let

  • block scoped

let x = 3;

if (true) {
    var x = 3;
    let y = 3;
}

x === 3

block scoped

var funcs = [];

for (let i = 0; i < 10; i++)
    funcs.push(function () { return i; });

funcs[0]() === 0

Destructuring

Destructuring

  • Object and Array literal-like syntax for assignment.
  • Declarative assignment.

Destructuring: Objects

let { a : a } = { a : 2 };
a === 2
// Desugared:
var b = { a : 2 }, a = b.a;
a === 2
let { a } = { a : 2 };
a === 2

Destructuring: Arrays

let [x] = [1];
x === 1
// Desugared:
var y = [1], x = y[0];
x === 1
let [a, , b, c] = [1, 2, 3, 4];
a === 1
b === 3
c === 4

The Rest Operator

The Rest Operator

  • Gathers 0+ elements from a matched list into a new array.
  • Allows for functional head/tail array destructuring.
let [head, ...tail] = [1, 2, 3, 4];
head === 1
tail === [2, 3, 4]
let [head, ...tail] = [];
head === undefined
tail === []
let [a, b, ...cs] = [1, 2, 3];
a  === 1
b  === 2
cs === [3]
