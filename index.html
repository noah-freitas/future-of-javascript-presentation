let

let

  • block scoped
  • lexically scoped
let x = 3;

block scoped

if (true) {
    var x = 3;
    let y = 3;
}

x === 3
y // ReferenceError: y is not defined

block scoped

var funcs = [];

for (let i = 0; i < 10; i++)
    funcs.push(function () { return i; });

funcs[0]() === 0
i // ReferenceError: i is not defined

lexically scoped

  • "temporal deadzone"
  • TypeErrors for redeclaration in blocks

lexically scoped

if (true) {
    x; // ReferenceError: can't access
       // let declaration `x' before initialization

    let x = 3;

    x = 5;

    let x = 4; // TypeError: redeclaration of variable x
    var x = 5; // TypeError: redeclaration of let x
}

Destructuring

Destructuring

  • Object and Array literal-like syntax for assignment.
  • Declarative assignment.

Destructuring: Objects

let { a : a } = { a : 2 };
a === 2
// Desugared:
var b = { a : 2 };
var a = b.a;
a === 2
let { a } = { a : 2 };
a === 2

Destructuring: Arrays

let [x] = [1];
x === 1
// Desugared:
var y = [1];
var x = y[0];
x === 1
